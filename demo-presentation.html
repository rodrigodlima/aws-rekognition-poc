OpenCV (Local Processing) 
import cv2

def detect_faces_opencv(image_path):
    # Load the image
    image = cv2.imread(image_path)
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # Load Haar Cascade classifier
    cascade_path = cv2.data.haarcascades + \
        'haarcascade_frontalface_default.xml'
    face_cascade = cv2.CascadeClassifier(cascade_path)

    # Detect faces with customizable parameters
    faces = face_cascade.detectMultiScale(
        gray,
        scaleFactor=1.1,
        minNeighbors=5,
        minSize=(30, 30)
    )

    print(f"Faces detected: {len(faces)}")

    # Draw rectangles and save
    for (x, y, w, h) in faces:
        cv2.rectangle(image, (x, y), (x+w, y+h),
                      (0, 255, 0), 2)

    cv2.imwrite('output.jpg', image)
    return faces
Run: python opencv_demo.py photo.jpg
AWS Rekognition (Cloud API) 
import boto3

def detect_faces_rekognition(image_path):
    rekognition = boto3.client('rekognition',
                               region_name='us-east-1')

    # Read image file
    with open(image_path, 'rb') as f:
        image_bytes = f.read()

    # Call Rekognition API - one line!
    response = rekognition.detect_faces(
        Image={'Bytes': image_bytes},
        Attributes=['ALL']
    )

    for face in response['FaceDetails']:
        bbox = face['BoundingBox']
        print(f"Position: {bbox['Left']:.2%}")
        print(f"Confidence: {face['Confidence']:.2f}%")

        # Rich attributes included!
        age = face.get('AgeRange', {})
        print(f"Age: {age.get('Low')}-{age.get('High')}")

        emotions = face.get('Emotions', [])
        top = max(emotions, key=lambda x: x['Confidence'])
        print(f"Emotion: {top['Type']}")

    return response
Run: python rekognition_demo.py photo.jpg
